145: // Mise en forme de la durée sous forme xj xxh xxm xxs
146: function pretty_time ($seconds) {
147:     $day = floor($seconds / (24 * 3600));
148:     $hs = floor($seconds / 3600 % 24);
149:     $ms = floor($seconds / 60 % 60);
150:     $sr = floor($seconds / 1 % 60);
151: 
152:     if ($hs < 10) { $hh = "0" . $hs; } else { $hh = $hs; }
153:     if ($ms < 10) { $mm = "0" . $ms; } else { $mm = $ms; }
154:     if ($sr < 10) { $ss = "0" . $sr; } else { $ss = $sr; }
155: 
156:     $time = '';
157:     if ($day != 0) { $time .= $day . 'j '; }
158:     if ($hs  != 0) { $time .= $hh . 'h ';  } else { $time .= '00h '; }
159:     if ($ms  != 0) { $time .= $mm . 'm ';  } else { $time .= '00m '; }
160:     $time .= $ss . 's';
161: 
162:     return $time;
163: }
